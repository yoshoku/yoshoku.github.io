def fit(x, _y = nil)
  check_sample_array(x)
  @norm_vec = Numo::NMath.sqrt((x**2).sum(1))
  self
end

def fit_transform(x, _y = nil)
  check_sample_array(x)
  fit(x)
  x / @norm_vec.tile(x.shape[1], 1).transpose
end

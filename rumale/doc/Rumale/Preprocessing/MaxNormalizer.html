Examples:

def fit(x, _y = nil)
  x = check_convert_sample_array(x)
  @norm_vec = x.abs.max(1)
  @norm_vec[@norm_vec.eq(0)] = 1
  self
end

def fit_transform(x, _y = nil)
  x = check_convert_sample_array(x)
  fit(x)
  x / @norm_vec.expand_dims(1)
end

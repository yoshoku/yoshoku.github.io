def randsvd(a, k, t = 0, driver: 'svd', job: 'A')
  n = a.shape[1]
  q = _orthonormal_mat(a, [k + 10, n].min, t)
  b = a.dot(q)
  s, u, vt = Numo::Linalg.svd(b, driver: driver, job: job)
  vtqt = vt.dot(q.transpose)
  _truncated_mat(s, u, vtqt, k)
end
